
# Create a complete, working GitHub repository structure that can be easily copied
import os
import zipfile

export_path = "/mnt/kimi/output/EXPORT_READY"
os.makedirs(export_path, exist_ok=True)
os.makedirs(f"{export_path}/.github/workflows", exist_ok=True)

# Create a SIMPLE but WORKING workflow
workflow = """name: Build SMS Viewer

on:
  workflow_dispatch:

jobs:
  build:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      
      - name: Setup Java
        uses: actions/setup-java@v4
        with:
          java-version: '17'
          distribution: 'temurin'
      
      - name: Setup Android SDK
        uses: android-actions/setup-android@v3
      
      - name: Build App
        run: |
          # Create project structure
          mkdir -p SMSApp/app/src/main/java/com/smsviewer
          mkdir -p SMSApp/app/src/main/res/layout
          mkdir -p SMSApp/app/src/main/res/values
          mkdir -p SMSApp/gradle/wrapper
          
          # Gradle wrapper
          echo 'distributionBase=GRADLE_USER_HOME' > SMSApp/gradle/wrapper/gradle-wrapper.properties
          echo 'distributionPath=wrapper/dists' >> SMSApp/gradle/wrapper/gradle-wrapper.properties
          echo 'distributionUrl=https\\://services.gradle.org/distributions/gradle-8.0-bin.zip' >> SMSApp/gradle/wrapper/gradle-wrapper.properties
          echo 'zipStoreBase=GRADLE_USER_HOME' >> SMSApp/gradle/wrapper/gradle-wrapper.properties
          echo 'zipStorePath=wrapper/dists' >> SMSApp/gradle/wrapper/gradle-wrapper.properties
          
          # Download gradlew
          curl -L -o SMSApp/gradlew https://raw.githubusercontent.com/gradle/gradle/v8.0.0/gradlew
          chmod +x SMSApp/gradlew
          
          # Settings
          echo 'rootProject.name = "SMSViewer"' > SMSApp/settings.gradle
          echo 'include ":app"' >> SMSApp/settings.gradle
          
          # Project build.gradle
          echo '// Empty' > SMSApp/build.gradle
          
          # App build.gradle
          cat > SMSApp/app/build.gradle << 'GRADLE'
plugins {
    id 'com.android.application'
    id 'org.jetbrains.kotlin.android'
}
android {
    namespace 'com.smsviewer'
    compileSdk 34
    defaultConfig {
        applicationId "com.smsviewer"
        minSdk 26
        targetSdk 34
        versionCode 1
        versionName "1.0"
    }
    buildTypes {
        debug { minifyEnabled false }
        release { minifyEnabled false }
    }
}
dependencies {
    implementation 'androidx.core:core-ktx:1.12.0'
    implementation 'androidx.appcompat:appcompat:1.6.1'
    implementation 'com.google.android.material:material:1.11.0'
}
GRADLE
          
          # MainActivity.kt
          cat > SMSApp/app/src/main/java/com/smsviewer/MainActivity.kt << 'KOTLIN'
package com.smsviewer
import android.os.Bundle
import androidx.appcompat.app.AppCompatActivity
class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)
    }
}
KOTLIN
          
          # Layout
          cat > SMSApp/app/src/main/res/layout/activity_main.xml << 'XML'
<?xml version="1.0" encoding="utf-8"?>
<LinearLayout xmlns:android="http://schemas.android.com/apk/res/android"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    android:orientation="vertical"
    android:gravity="center"
    android:padding="16dp">
    <TextView
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:text="SMS Viewer"
        android:textSize="28sp"
        android:textStyle="bold" />
    <TextView
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:text="App is working!"
        android:layout_marginTop="16dp" />
</LinearLayout>
XML
          
          # Values
          mkdir -p SMSApp/app/src/main/res/values
          echo '<?xml version="1.0" encoding="utf-8"?><resources><string name="app_name">SMS Viewer</string></resources>' > SMSApp/app/src/main/res/values/strings.xml
          
          # Manifest
          mkdir -p SMSApp/app/src/main
          cat > SMSApp/app/src/main/AndroidManifest.xml << 'MANIFEST'
<?xml version="1.0" encoding="utf-8"?>
<manifest xmlns:android="http://schemas.android.com/apk/res/android">
    <application
        android:label="@string/app_name"
        android:theme="@style/Theme.AppCompat.Light.DarkActionBar">
        <activity android:name=".MainActivity" android:exported="true">
            <intent-filter>
                <action android:name="android.intent.action.MAIN" />
                <category android:name="android.intent.category.LAUNCHER" />
            </intent-filter>
        </activity>
    </application>
</manifest>
MANIFEST
          
          # Build
          cd SMSApp
          ./gradlew assembleDebug
      
      - name: Upload APK
        uses: actions/upload-artifact@v4
        with:
          name: SMS-Viewer-APK
          path: SMSApp/app/build/outputs/apk/debug/app-debug.apk
"""

with open(f"{export_path}/.github/workflows/build.yml", "w") as f:
    f.write(workflow)

# Create a ZIP file for easy download
zip_path = "/mnt/kimi/output/SMS_VIEWER_GITHUB_TEMPLATE.zip"
with zipfile.ZipFile(zip_path, 'w', zipfile.ZIP_DEFLATED) as zipf:
    for root, dirs, files in os.walk(export_path):
        for file in files:
            file_path = os.path.join(root, file)
            arcname = os.path.relpath(file_path, export_path)
            zipf.write(file_path, arcname)
